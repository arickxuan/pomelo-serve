## 2. Docker镜像制作

### 2.1 user服务

~~~dockerfile
FROM registry.cn-hangzhou.aliyuncs.com/mszlu/go:1.21 AS build-stage
WORKDIR /app
COPY . ./
RUN go env -w GO111MODULE=on
RUN go env -w GOPROXY=https://goproxy.cn,direct
RUN go mod download
RUN CGO_ENABLED=0 GOOS=linux go build  -o /app/user/user user/main.go
FROM registry.cn-hangzhou.aliyuncs.com/mszlu-gcrio/distroless_base-debian11 AS build-release-stage
WORKDIR /app
COPY --from=build-stage /app/user/user /app/user
EXPOSE 11500
USER nonroot:nonroot
ENTRYPOINT ["/app/user"]
~~~

## 3. 上传harbor

将镜像上传harbor，方便使用

~~~shell
docker login -u admin -p Harbor12345 testharbor.com
docker push testharbor.com/qp/user:1.0
~~~

windows:

`C:\Users\Huo\.docker`

daemon.json

~~~json
{
  "builder": {
    "gc": {
      "defaultKeepStorage": "20GB",
      "enabled": true
    }
  },
  "experimental": false,
  "features": {
    "buildkit": true
  },
  "insecure-registries":["testharbor.com"]
}

~~~

config.json:

~~~json
{
	"auths": {
		"https://index.docker.io/v1/": {},
		"https://testharbor.com/v2/": {
			"auth": "YWRtaW46SGFyYm9yMTIzNDU="
		},
		"registry.cn-hangzhou.aliyuncs.com": {},
		"registry.cn-shenzhen.aliyuncs.com": {},
		"testharbor.com": {}
	},
	"credsStore": "desktop"
}
~~~

创建`certs.d/testharbor.com`目录，把证书下载到此目录

## 4. 部署k8s

### 4.1 ingress

~~~yaml
apiVersion: networking.k8s.io/v1
kind: Ingress
metadata:
  name: qp-ingress-http
  namespace: qp
  annotations:
    nginx.ingress.kubernetes.io/proxy-body-size: "50m"
    nginx.ingress.kubernetes.io/Access-Control-Allow-Origin: '*'
    nginx.ingress.kubernetes.io/cors-allow-headers: "DNT,X-CustomHeader,Keep-Alive,User-Agent,X-Requested-With,If-Modified-Since,Cache-Control,Content-Type,Authorization,token"
    nginx.ingress.kubernetes.io/cors-allow-methods: 'PUT, GET, POST, OPTIONS'
    nginx.ingress.kubernetes.io/cors-allow-origin: '*'
    nginx.ingress.kubernetes.io/service-weight: ''
    nginx.org/websocket-services: "qp-connector-service"
    nginx.ingress.kubernetes.io/configuration-snippet: |
       proxy_set_header Upgrade "websocket";
       proxy_set_header Connection "Upgrade";
spec:
  ingressClassName: "nginx"
  rules:
    - host: qp.test.com
      http:
        paths:
          - path: /
            pathType: Prefix
            backend:
              service:
                name: qp-gate-service
                port:
                  number: 13000
    - host: qpws.test.com
      http:
        paths:
          - path: /
            pathType: Prefix
            backend:
              service:
                name: qp-connector-service
                port:
                  number: 12000
~~~

~~~yaml
apiVersion: apps/v1
kind: Deployment
metadata:
  name: qp-user-deployment
  namespace: qp
  labels:
    app: user
spec:
  selector:
    matchLabels:
      app: user
  replicas: 1
  minReadySeconds: 5
  strategy:
    type: RollingUpdate
    rollingUpdate:
      maxSurge: 1
      maxUnavailable: 1
  template:
    metadata:
      labels:
        app: user
    spec:
      containers:
        - image: testharbor.com/qp/user:1.0
          name: user
          imagePullPolicy: Always
          args: ["--config", "/app/conf/application.yml"]
          ports:
            - containerPort: 11500
              protocol: TCP
          volumeMounts:
            - mountPath: /app/conf/
              name: user-config
      volumes:
        - name: user-config
          configMap:
            name: user-conf
---
apiVersion: v1
kind: Service
metadata:
  name: user
  namespace: qp
spec:
  selector:
    app: user
  type: ClusterIP
  ports:
    - port: 11500
      targetPort: 11500
~~~
