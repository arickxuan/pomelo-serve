func (l *Logic) getCards(num int) []CardID {
	l.Lock()
	defer l.Unlock()
	if len(l.cards) < num {
		return nil
	}
	cards := l.cards[:num]
	l.cards = l.cards[num:]
	return cards
}

func (l *Logic) getRestCardsCount() int {

	count := 0

		if v == Zhong {

	return count
